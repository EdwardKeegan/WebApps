ES6 Destructuring (Objects and Arrays). Press F12 (open JS console) to see output. 

    // ES6 object destructuring (assigns by property name, not position)
    let {b, a} = {a: "apple", b: "banana", c:"carrot"};     
    console.log("a is " + a + " and b is " + b); // prints: a is apple and b is banana

    // ES6 array destructuring (assigns by position)
    let [ d, f, g ] = [ "dog", "frog", "goat", "elephant" ];             
    console.log("d is " + d + ", f is " + f + 
         ", and g is " + g); // prints: d is dog, f id frog, and g is goat
