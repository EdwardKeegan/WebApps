Demonstrating For Each - Press F12 to See Output

Iterating over each element in an array without needing a "for loop". 

    const carList = [
        {id: 1, name: "Duster", year: 1980},
        {id: 2, name: "Accord", year: 2019},
        {id: 3, name: "Prius", year: 2010}
    ];

    console.log("for each - old way");
    for (let i = 0; i < carList.length; i++) {
        console.log(carList[i].name);
    }

    console.log("for each - functional programming (anonymous function)");
    carList.forEach(function (car) {
        console.log(car.name);
    });

    console.log("named function");
    carList.forEach(function doOne(car) {
        console.log(car.name);
    });

    console.log("named function expression");
    carList.forEach( doOne = function (car) {
        console.log(car.name);
    });

    console.log("arrow function");
    carList.forEach(doOne = (car) => {
        console.log(car.name);
    });

    console.log("anonymous arrow function");
    carList.forEach((car) => {
        console.log(car.name);
    });

    console.log("abbreviated arrow function");
    carList.forEach(car => console.log(car.name));
