    function Person(first, last, birthYear) {
        this.firstName = first;
        this.lastName = last;
        this.birthYear = birthYear;
        this.setBirthYear = function (newBirthYear) {
            this.birthYear = newBirthYear;
        };
        this.show = function () {
            return this.firstName + " " + this.lastName + " was born in " + this.birthYear;
        };
    }

    var p1 = new Person("sally", "key", 1980);
    var p2 = new Person("joe", "smith", 1970);
    console.log(p1.show());
    console.log(p2.show());
    p1.setBirthYear(2012);
    console.log(p1.show());
