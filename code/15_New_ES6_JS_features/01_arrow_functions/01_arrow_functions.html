ES6 Arrow Functions. Press F12 (open JS console) to see output. 

 
    // Function expression definition in "regular JavaScript" (called ES5).
    var multiply_ES5 = function (x, y) {
        return x * y;
    };

    // Can use ES6 const instead of var (since we won't redefine the function later)
    const multiply_ES5 = function (x, y) {
        return x * y;
    };

    // New (ES6) Arrow Function syntax: don't need the keyword "function"
    const multiply_ES6 = (x, y) => {
        return x * y;
    };

    // ES6: don't need curly brackets if the arrow function just returns an expression.
    const multiply_ES6_v2 = (x, y) => x * y;

    // ES6 syntax - OK to have more than one line in the arrow function 
    const multiply_ES6_v3 = (x, y) => {
        console.log("Yes, it is ok to have more than one line in the function.");
        return x * y;
    };

    // output
    console.log("multiply_ES5(2,3) is " + multiply_ES5(2, 3));
    console.log("multiply_ES6(3,4) is " + multiply_ES6(3, 4));
    console.log("multiply_ES6_v2(4,5) is " + multiply_ES6_v2(4, 5));
    console.log("multiply_ES6_v3(5,6) is " + multiply_ES6_v3(5, 6));
