    function MakePerson(first, last, birthYear) {
        var person = {};
        person.firstName = first;
        person.lastName = last;
        person.birthYear = birthYear;
        person.setBirthYear = function (newBirthYear) {
            person.birthYear = newBirthYear;
        };
        person.show = function () {
            return person.firstName + " " + person.lastName + " was born in " + person.birthYear;
        };
        return person;
    }

    var p1 = MakePerson("sally", "key", 1980);
    var p2 = MakePerson("joe", "smith", 1970);
    console.log(p1.show());
    console.log(p2.show());
    p1.setBirthYear(2012);
    console.log(p1.show());
